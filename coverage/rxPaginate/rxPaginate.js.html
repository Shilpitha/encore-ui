                $scope.showLoadingOverlay = false;
            };
 
            this.showAndHide = function (promise) {
                this.show();
                promise.finally(this.hide);

            $compile(loadingBlockHTML)(scope, function (clone) {
                element.append(clone);
            });
        }
    };
